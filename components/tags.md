<script>
    // Tag component showcase functionality
    document.addEventListener('DOMContentLoaded', function () {
        // Background toggle functionality for demo containers
        const toggleButtons = document.querySelectorAll('.demo-toggle-btn');

        toggleButtons.forEach(button => {
            button.addEventListener('click', function () {
                const toggleData = JSON.parse(this.getAttribute('data-toggler'));
                const [className, targetSelector, stateData] = toggleData;
                const targets = document.querySelectorAll(targetSelector);

                targets.forEach(target => {
                    target.classList.toggle(className);
                });

                // Update button text
                const label = this.querySelector('.label');
                if (this.textContent.includes('Remove')) {
                    label.textContent = label.textContent.replace('Remove', 'Add');
                } else if (this.textContent.includes('Add')) {
                    label.textContent = label.textContent.replace('Add', 'Remove');
                } else if (this.textContent.includes('Dark')) {
                    label.textContent = 'Light bg';
                } else if (this.textContent.includes('Light')) {
                    label.textContent = 'Dark bg';
                }
            });
        });
    });
</script>
